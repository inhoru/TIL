# 1. 배열
1. js의 배열은 java List와 유사하다.<br/>
2. 배열저장소에는 다양한 type의 값을 저장할 수 있다.<br/>
3. 인덱스번호로 배열에 있는 값을 출력하고, 배열에 값을 저장할 수 있다.<br/>
4. 길이제한이 없고, 자동으로 증가함 * 공간에 대해 미리 설정할 필요가 없다.<br/>
5. js배열은 Array오브젝트로 편의 메소드(함수)를 제공한다.

<br/>

```javascript
<script>
        //배열 선언하기
        //1. 리터럴로 선언하기 : [],[값,값,값,값......]
        //2. 생성자로 선언하기 : new Array(), new Array(값,값,값,값,....)
        let arr=[];
        let arr2=new Array();
        console.log(arr,typeof arr);
        console.log(Object.prototype.toString.call(arr));
        console.log(arr2,typeof arr2);
        console.log(Object.prototype.toString.call(arr2));
        //ES6 배열여부확인 하기
        console.log(Array.isArray(arr));
        console.log(Array.isArray(10));

        //생성과 동시에 초기화하기
        arr=[1,2,3,4,5];
        arr2=new Array('가','나','다','라','마');
        console.log(arr);
        console.log(arr2);
        arr=['가',10,true,[1,2,3,4],{name:'유병승',age:19},()=>{console.log('배열내부함수')}];
        console.log(arr);

        //배열에 값 대입하기
        arr[0]='유병승';
        arr[1]=19;
        arr[2]=["키보드","모니터","최주영"];
        arr[3]=(msg)=>console.log(msg);
        console.log(arr);

        //배열에 들어간 함수호출하기
        arr[3]("여러분 화이팅!!!");
        console.log(arr[2][0]);

        //길이의 제한이 없다.
        arr=[];
        console.log(arr);
        arr[0]='김허녕 힘내! 할수있따';
        console.log(arr[0]);
        arr[1]=100;
        console.log(arr);
        arr[10]=180.5;
        console.log(arr);
        console.log(arr[8]);
        console.log(arr[13]);
        console.log(arr);


        //배열의 길이를 확인하기
        console.log(arr.length);

        //반복문을 이용해서 배열에 있는값 순회하기
          for(let i=0;i<arr.length;i++){
              console.log(arr[i]);   
          }

        //for of문 이용하기 -> for(변수 : 배열) 유사
        for(let e of arr){
            console.log(e);
        }
</script>
```
